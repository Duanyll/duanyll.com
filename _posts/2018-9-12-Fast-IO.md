别忘了 `#include`

```cpp
#include<cstdio>
#include<cctype>
```

```cpp
inline int read()
{
    int X = 0, w = 0;
    char ch = 0;
    while (!isdigit(ch))
    {
        w |= ch == '-';
        ch = getchar();
    }
    while (isdigit(ch))
        X = (X << 3) + (X << 1) + (ch ^ 48), ch = getchar();
    return w ? -X : X;
}
```

```cpp
inline double dbread()
{
    double X = 0, Y = 1.0;
    int w = 0;
    char ch = 0;
    while (!isdigit(ch))
    {
        w |= ch == '-';
        ch = getchar();
    }
    while (isdigit(ch))
        X = X * 10 + (ch ^ 48), ch = getchar();
    ch = getchar(); 
    while (isdigit(ch))
        X += (Y /= 10) * (ch ^ 48), ch = getchar();
    return w ? -X : X;
}
```

```cpp
inline void write(int x)
{
    if (x < 0)
        putchar('-'), x = -x;
    if (x > 9)
        write(x / 10);
    putchar(x % 10 + '0');
}
```
